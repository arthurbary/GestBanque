//cree la bibliotheque
dotnet new classlib -o Models
//rajoute la  bibliotheque a la solution ( peut aussi rajouter un projet ou autre elements)
dotnet sln add .\Models
//Pour ajouter des package (nugette) ou rajouter reference
dotnet add .\ProjectName reference .\Models
dotnet build
